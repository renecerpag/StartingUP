// GRAPH

let chart;

async function loadChart() {

  const ctx = document.getElementById('stockChart').getContext('2d');

  const response = await fetch("https://www.alphavantage.co/query?function=TIME_SERIES_DAILY&symbol=AAPL&apikey=F08ZQ7OG7SR35UO1");

  const data = await response.json();

  const series = data["Time Series (Daily)"];

  const labels = Object.keys(series).slice(0, 10).reverse();

  const values = labels.map(d => series[d]["4. close"]);

  if (chart) chart.destroy();

  chart = new Chart(ctx, {

    type: 'line',

    data: { labels, datasets: [{ label: "AAPL", data: values, borderColor: 'gold' }] },

  });

}

loadChart();

</script>
